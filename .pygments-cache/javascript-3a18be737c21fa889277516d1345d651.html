$(function() {

	$("#touch").on("click", function() {
		// Ajax通信
		ajax();
	});

	function ajax() {
		$.ajax({
			type: "POST",
			url: "http://localhost:9999/1",
			dataType: "json"
		}).done(function(data) {
			alert(data.status + "\n" + data.description);
		}).fail(function(data) {
			alert("error!!");
		});
	}
});
