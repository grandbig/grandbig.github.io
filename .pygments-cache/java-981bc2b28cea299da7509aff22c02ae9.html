// Skill.java
public class Skill extends RealmObject {

	@PrimaryKey
	private int id;
	private String skill;

	public int getId() {
		return id;
	}
	public void setId(int id) {
		this.id = id;
	}

	public String getSkill() {
		return skill;
	}
	public void setSkill(String skill) {
		this.skill = skill;
	}
}
