// CounterReducers.swift
struct CounterReducer: Reducer {
	func handleAction(action: Action, state: AppState?) -> AppState {
		var state = state ?? AppState()

		switch action {
			case _ as CounterActionIncrease:
				state.counter += 1
			case _ as CounterActionDecrease:
				state.counter -= 1
			default:
				break
		}

		return state
	}
}
