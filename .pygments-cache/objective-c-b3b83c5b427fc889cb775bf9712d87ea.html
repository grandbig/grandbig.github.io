// Company.swift
import Foundation

protocol Person {
    func hire() -> String
}

class CompanyOwner: Person {
    let humanResource: Engineer

    init(humanResource: Engineer) {
        self.humanResource = humanResource
    }

    func hire() -> String {
        let name = humanResource.name ?? "someone"
        return "I'm hiring \(name). \(humanResource.develop())"
    }
}
