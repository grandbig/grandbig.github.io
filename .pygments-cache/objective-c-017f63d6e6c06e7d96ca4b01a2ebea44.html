func sample1() {
  let chrSample1 = "こんにちは"
  // 先頭文字の取得
  print(chrSample1.characters[chrSample1.startIndex])
  // 先頭から1つ先の文字を取得
  let secondIndex = chrSample1.index(after: chrSample1.startIndex)
  print(chrSample1.characters[secondIndex])
  // 末尾の文字を取得
  let lastBeforeIndex = chrSample1.index(before: chrSample1.endIndex)
  print(chrSample1.characters[lastBeforeIndex])
}
