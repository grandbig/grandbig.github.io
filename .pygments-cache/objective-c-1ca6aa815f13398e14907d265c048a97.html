- (void)viewDidLoad {
	[super viewDidLoad];

	// インスタンス化
	[SingletonSample sharedInstance];
	// インスタンス変数を確認
	NSLog(@"num: %ld, str: %@", [SingletonSample sharedInstance].num, [SingletonSample sharedInstance].str);
	
	// クラスメソッドの実行
	[SingletonSample testClassMethod];
	// インスタンスメソッドの実行
	[SingletonSample sharedInstance] testInstanceMethod];
	
	// インスタンス変数
	NSLog(@"num: %ld, str: %@", [SingletonSample sharedInstance].num, [SingletonSample sharedInstance].str);]
}
