// reducers/index.js
import { handleActions } from 'redux-actions'

const initialState = {
  value: 0,
  num: 0    // 追加 (1)
};

export default handleActions({
  INCREMENT: (state, action) => {
    const newState = {
      ...state,   // 追加 (2)
      value: state.value + 1
    };
    return newState;
  },
  DECREMENT: (state, action) => {
    const newState = {
      ...state,   // 追加 (2)
      value: state.value - 1
    };
    return newState;
  },
  // 以下が追加 (3)
  SHOW_TOAST: (state, action) => {
    const newState = {
      ...state,
      num: state.num + 1
    };
    return newState;
  }
}, initialState)
